### if, if else, if else if
```swift
// if
let age = 12

if age < 19 {           // true
    print("미성년자")     // "미성년자"
}

// if else
let age = 20

if age < 19 {          // false
    print("미성년자")    // pass
} else {
    print("성인")       // "성인"
}

// if else if
let animal = "cat"

if animal === "dog" {           // false
    print("강아지 사료 주기")       // pass
} else if animal === "cat" {    // true
    print("고양이 사료 주기")       // "고양이 사료 주기"
} else {                        // 앞 조건문 ture로 pass 
    print("해당하는 동물 사료 없음")
}
```
### switch
```swift
// 특정 값
let color = "green"

switch color {
case "blue":
    print("파란색 입니다.") // pass

case "green":
    print("초록색 입니다.") // "초록색 입니다."

case "yellow":
    print("노란색 입니다.") // pass

default:
    print("찾는 색상이 없습니다.") // blue, green, yellow 중 해당 없으면 출력
}

// 범위 연산자
let temperature = 30

switch temperature {
case -20...9:
    print("겨울 입니다.")

case 10...14:
    print("가을 입니다.")

case 15...25:
    print("봄 입니다.")

case 26...35:
    print("여름 입니다.")

default:
    print("이상 기후입니다.")
}
```
